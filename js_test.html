5. What will be the output of the following code involving `Object.create()`?

function A() {
    this.name = 'A';
}
A.prototype.sayHello = function() {
    return 'Hello from A';
};

function B() {
    A.call(this);
    this.name = 'B';
}
B.prototype = Object.create(A.prototype);
B.prototype.constructor = B;

const b = new B();
console.log(b.sayHello());
console.log(b instanceof A);
console.log(b instanceof B);

12. What will be the output of the following code with method chaining?

class Counter {
    constructor() {
        this.count = 0;
    }
    increment() {
        this.count += 1;
        return this;
    }
    decrement() {
        this.count -= 1;
        return this;
    }
    getValue() {
        return this.count;
    }
}
const counter = new Counter();
const result = counter.increment().increment().decrement().getValue();
console.log(result);
